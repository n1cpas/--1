#include <stdio.h>

#define MOD 12345

int main() {
    int n;
    printf("Введіть довжину послідовності n: ");
    scanf("%d", &n);

    if (n <= 1 || n >= 10000) {
        printf("Введене число n не відповідає обмеженням (1 < n < 10000).\n");
        return 1; // Повертаємо код помилки
    }

    long long dp[n + 1][2]; // Масив для збереження кількості послідовностей для кожного стану

    // Ініціалізація для довжини послідовності 1
    dp[1][0] = 1;
    dp[1][1] = 1;

    // Обчислення кількості послідовностей для довжини 2 до n
    for (int i = 2; i <= n; i++) {
        dp[i][0] = (dp[i - 1][0] + dp[i - 1][1]) % MOD;
        dp[i][1] = dp[i - 1][0];
    }

    long long total = (dp[n][0] + dp[n][1]) % MOD;
    printf("Кількість шуканих послідовностей: %lld\n", total);

    return 0;
}
